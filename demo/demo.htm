  Some content

instructions

configuration

use the appropriate selector to find your windows, and call the windows() method

<script src="jquery.windows.js"></script>

<script>
$('.window').windows({
  snapping: true,     // Turn on snapping
  snapSpeed: 500,     // How quickly to snap to a window
  snapInterval: 1100, // Delay after scroll completes before snapping starts
  onScroll: function(scrollPos){
    // scrollPos:Number
  },
  onSnapComplete: function($el){
    // after window ($el) snaps into place
  },
  onWindowEnter: function($el){
    // when new window ($el) enters viewport
  }
});
</script>

api

// returns ratio 0-1
$('.window:eq(0)').ratioVisible();

// returns boolean of whether or not is visible on screen
$('.window:eq(0)').isOnScreen();

// get most visible window
$('.window').getCurrentWindow();
